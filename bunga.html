Kode Python untuk Menggambar Bunga


import turtle

# Set up the screen
screen = turtle.Screen()
screen.bgcolor("lightblue")  # Light blue background

# Create a turtle object for drawing
t = turtle.Turtle()
t.speed(5)

# Function to draw a petal
def draw_petal():
    t.color("gold")
    t.begin_fill()
    t.circle(70, 60)  # First half of the petal (bigger petal)
    t.left(120)
    t.circle(70, 60)  # Second half of the petal
    t.left(120)
    t.end_fill()

# Function to draw the flower
def draw_flower():
    # Draw multiple petals
    for _ in range(12):  # More petals for a full look
        draw_petal()
        t.right(30)  # Rotate to position for the next petal

# Function to draw the center of the sunflower
def draw_center():
    t.penup()
    t.goto(30, 0)  # Center position relative to the flower
    t.pendown()
    t.color("chocolate")
    t.begin_fill()
    t.circle(30)  # Center of the flower
    t.end_fill()

# Function to draw the stem
def draw_stem():
    t.penup()
    t.goto(0, -300)  # Position the turtle at the bottom for the stem
    t.setheading(90)  # Point upwards for the stem
    t.pendown()
    t.color("forestgreen")  # Stem color
    t.pensize(10)  # Stem thickness
    t.forward(300)  # Stem length

# Function to draw leaves
def draw_leaf(x, y):
    t.penup()
    t.goto(x, y)
    t.setheading(-45)
    t.pendown()
    t.color("forestgreen")
    t.begin_fill()
    t.circle(50, 90)  # Half of the leaf
    t.left(90)
    t.circle(50, 90)  # Other half of the leaf
    t.end_fill()

# Draw the stem
draw_stem()

# Move the turtle to draw the flower
t.penup()
t.goto(0, 0)  # Position for the flower
t.pendown()
draw_flower()

# Draw the center of the sunflower
draw_center()

# Draw leaves on the stem
draw_leaf(-75, -150)  # Left leaf
draw_leaf(0, -150)   # Right leaf

# Hide the turtle and display the result
t.hideturtle()
screen.mainloop()
